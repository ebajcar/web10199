    const email = document.getElementById("mail");
function customizeValidityMessages(event) {
  if (email.validity.typeMismatch) {
    email.setCustomValidity("This does not look like a valid email!" + 
        "\nHow long can you make this message? \nYou can break a line with \\n");
  } else {
    email.setCustomValidity("OK, that may be valid! \nThis is modified too.");
  }
}
email.addEventListener("input", customizeValidityMessages);
    Custom messages are displayed, both when input is invalid and valid.
