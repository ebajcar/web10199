<script>
// rest parameters
function multiply3(multiplier, ...theArgs) {
  return theArgs.map(x => multiplier * x);
}
let arr = multiply3(2, 1, 2, 3);
let display = document.getElementsByTagName("output")[0];
display.innerText = "Output 1: " + arr + "\n"; // [2,4,6]
arr = multiply3(7, 1, 2, 3);
display.innerText += "Output 2: " + arr + "\n"; // [7,14,21]

function powerOfThree(power, ...theArgs) {
  return theArgs.map(x => x ** power);
}
arr = powerOfThree(3, 1, 2, 3);
display.innerText += "Output 3: " + arr + "\n"; // [1,8,27]
arr = powerOfThree(4, 1, 2, 3);
display.innerText += "Output 4: " + arr + "\n"; // [1,16,81]  
</script>
