<script>
// default parameters
function multiply1(a, b) {
  b = typeof b !== 'undefined' ? b : 1; // you could test 
  return a * b;
}
// better using default parameters
function multiply2(a, b = 1) {
  return a * b;
}
document.getElementsByTagName("output")[0].innerText = 
            "Output 4A: using multiply2(5) " + multiply2(5) + "\n";
document.getElementsByTagName("output")[0].innerText += 
            "Output 4B: using multiply2(5,3) " + multiply2(5, 3) + "\n";
</script>
