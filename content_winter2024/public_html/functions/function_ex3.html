<script>
// arguments[i]
function myConcat(separator) {
  let result = ''; // initialize list
  let i, len = arguments.length;
  // iterate through arguments
  for (i = 1; i < len; i++) {
    result += arguments[i] + separator;
  }
  return result;
}
let display = document.getElementsByTagName("output")[0];
// returns "red, orange, blue, "
display.innerText = "Output 3A: " + myConcat(', ', 'red', 'orange', 'blue') + "\n";
// returns "elephant; giraffe; lion; cheetah; "
display.innerText += "Output 3B: " + myConcat('; ', 'elephant', 'giraffe', 'lion', 'cheetah') + "\n";
// returns "sage. basil. oregano. pepper. parsley. "
display.innerText += "Output 3C: " + 
                      myConcat('. ', 'sage', 'basil', 'oregano', 'pepper', 'parsley') + "\n";
</script>
