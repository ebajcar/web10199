function functionForIn() {
  let person = {
    fname: "John", 
    lname: "Doe", 
    age: 25,
    gender: "male"
  };   
  let text = "";
  let prop;
  for (prop in person) {
    text += person[prop] + " ";
  }
  return text;
}

  • The for…in structure takes each property in an Object and returns the value of the property.
