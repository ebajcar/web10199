traditional function arrow function methods

  • Using traditional function

    let points = [
  {x:0, y:0},
  {x:1, y:1}
];
// creating custom methods using traditional function
points.dist = function () {  	
let p1 = this[0];
let p2 = this[1];
let a = p2.x - p1.x;
let b = p2.y - p1.y;
return Math.sqrt( a * a + b * b );
};
handl.innerText = points.dist(); 
}
    Output

  • Using arrow function

    let points2 = [
  {x:0, y:0},
  {x:1, y:1}
];
// An arrow function expression is a compact alternative to a traditional function
// expression, but is limited and can't be used in all situations.
// creating custom methods using arrow function
points2.dist2 = () => { 
    let p1 = this[0];
    let p2 = this[1];
    let a = p2.x - p1.x;
    let b = p2.y - p1.y;
    return Math.sqrt( a * a + b * b );
};
handl.innerText = points2.dist2();
    Output
