    // list each property within the DOM window object
const panel = document.getElementsByClassName("apanel") ; 
let property; 
let count1 = 0;
for ( property in window.document.body ) { 
  if ( property ) { 
    panel[0].innerHTML += property + "—" ; 
    count1++;
  }
}
panel[0].innerHTML += "<h3>There are " + count1 + 
                      " properties in the window.document.body object.</h3>";
let count2 = 0;
for ( property in window ) { 
  if ( property ) { 
    panel[1].innerHTML += property + " — " ; 
    count2++;
  }
}
panel[1].innerHTML += "<h3>There are " + count2 + 
                      " properties in the window object.</h3>";
