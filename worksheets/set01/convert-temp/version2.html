let fahrenheit = 90;
let celsius =  (fahrenheit - 32) * 5 / 9 ;
document.write("<p>" + fahrenheit + 
  "°F temparature is ", celsius.toFixed(3), "°C</p>" );
document.write(fahrenheit + "°F temparature is " + 
                celsius.toPrecision(4) + "°C" );

NOTE: document.write() appends to the end of the existing content
