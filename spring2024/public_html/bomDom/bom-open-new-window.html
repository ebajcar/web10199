document.querySelect("button").addEventListener("click", function() {	
        let pop = window.open( "", "MyPopup", 
                    "top=200,left=100,width=450,height=150,resizable,scrollbars=no,status=1");
        pop.document.write( "<title>Changed value Popup Window</title>" ) ; 
        pop.document.write( "<h1>Web Programming Example</h1>" ); 
        pop.document.write( "<p>Dynamic Popup Content</p>" ); 
        pop.document.close();
    }, 
    false
);
