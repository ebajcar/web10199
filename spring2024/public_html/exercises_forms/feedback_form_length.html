    function bindToMaxLength(max=200) {
    const leftover = document.getElementById("feedback");
    leftover.textContent = max;
    const field = document.getElementsByClassName('maxLength')[0]; 
	field.addEventListener("keyup", function() {
        let curr = field.value;
        let clen = curr.length;
        (clen < max) ? delta = max - clen : delta = 0;
        leftover.innerText = delta;
        if (curr > max) 
            leftover.innerText = "";
    });
}
