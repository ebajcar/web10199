Using DOM methods, we create a table with rows and columns and populate data from the custom object named "colours".

<script>
const colours = {
  black:  "000000",
  green:  "008000",
...
  aqua:   "00FFFF",
  fuchsia:"FF00FF"
};

function createColorTable(obj) {
  let parent = document.getElementsByTagName("table")[0];
  parent.appendChild(document.createElement("caption")).innerText = 
        "Color names and RGB values";
  let index = 0, td1, td2, tr;
  for (let property in obj) {
      if (!(index % 2)) {
        tr = parent.appendChild(document.createElement("tr")); 
      }
      index++;
      td1 = tr.appendChild(document.createElement("td"));
      td1.style.backgroundColor = property;     
      td2 = tr.appendChild(document.createElement("td"));  
      td2.innerText = property + "= #" + obj[property];
  }
}
createColorTable(colours);

</script>
