Array object and Function objects - explore properties and methods

Using an earlier example (array-random-onclick.html), the code is refactored so the code can be re-used for any array by passing the name of the array to a function. This example uses event listeners to display the result.

Compare using event methods (array-random-function.html) and event listeners (array-random-listener.html).

  • Click anywhere to generate next random name and display it here.

  • Example functions

    
"use strict";
// generic random number generator
function random(min, max) {
  return Math.floor(Math.random() * (max - min)) + min;
}

// calls the function random()
// returns the randomly picked member from array names
function chooseItem(array, len) {
  return array[random(0, len)];
}

const names = ['Chris', 'Li Kang', 'Anne', 'Francesca', 'Mustafa', 'Tina', 'Bert', 'Jada'];
const numbers = [ 343, 333, 234, 1001, 23413, 234234, 1, 34, 534, 98 ];

function displayChoice(arr) {
  let h3 = document.createElement('h3');
  h3.textContent = chooseItem(arr, arr.length);
  const section = document.querySelector('section');
  //section.innerHTML = "";  // removes existing sub-structure
  section.appendChild(h3);
  console.log(section.innerHTML);
}

document.getElementsByTagName("body")[0].onclick = function () { displayChoice(numbers); };
document.getElementsByTagName("body")[0].onclick = function () { displayChoice(names); };
