Date object - date range

Date standard built-in object — MDN docs and Javascript.info
You can interact with the timestamp value directly using the getTime() and setTime() methods.
All static methods (Date.now(), Date.parse(), and Date.UTC()) return timestamps instead of Date objects.
The Date() constructor can be called with a timestamp as the only argument.
A date is represented internally as a single number, the timestamp.

<script>
"use strict";
const all_tags = document.getElementsByTagName("*");

function showDays() {
    let allTags = document.getElementsByTagName("*");
    let len = allTags.length;
    for (let i=0;i < len; i++) {
        if (allTags[i].className.indexOf("daysTill") > -1) {
            //console.log("className.indexOf " + showTheDaysTill(allTags[i].id));
            allTags[i].innerText = showTheDaysTill(allTags[i].id);
            //console.log("allTags " + allTags[i].className);
        }
    }
}  
  
function showTheDaysTill(thisDate) {
    let theDays;
    switch(thisDate) {
        case "readweek":
            theDays = daysTill(6,21);
            break;
        case "endterm":
            theDays = daysTill(8,23);
            break;					
        default:
    }
    return theDays + " ";
}

function daysTill(mm,dd) {
    let now = new Date();
    let inDate = new Date(now.getFullYear(), mm-1,dd);
    if (inDate.getTime() < now.getTime()) {
        inDate.setYear(now.getFullYear()+1);
    }
    return (Math.ceil(dayToDays(inDate) - dayToDays(now)));
}

function dayToDays(inTime) {
    return (inTime.getTime() / (1000 * 60 * 60 * 24));
}

document.addEventListener("DOMContentLoaded", () => {
    showDays();
    today.innerText = new Date();
    today.innerHTML += "<br>Output of Date.now() - timestamp: " + Date.now();
});
</script>
