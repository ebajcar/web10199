    // generic random number generator
function random(min, max) {
  return Math.floor(Math.random() * (max - min)) + min;
}
// array random function
// calls the function random()
// returns the randomly picked member of array
function chooseItem(array, len) {
  return array[random(0, len)];
}

document.getElementsByTagName("body")[0].onclick = function () {
  const names =
    ['Chris', 'Li Kang', 'Anne', 'Francesca', 'Mustafa', 'Tina', 'Bert', 'Jada'];
  let h3 = document.createElement('h3');
  h3.textContent = chooseItem(names, names.length);
  console.log(h3.textContent);
  const section = document.querySelector('section');
  section.innerHTML = "";
  section.appendChild(h3);
}
